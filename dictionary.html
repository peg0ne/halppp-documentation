Dictionary

Dictionary is class with key value pairs

Constructors

Dictionary<T, U> => Dictionary<T, U>

Functions

keys => vector<T>
contains T key => bool
add T key, U value => void
try_get_value T key => Option<U>

Examples

let dict = Dictionary<string, int>() //initialize dictionary

dict.add("one", 1) //adds key value pair "one" -> 1
dict.add("two", 2) //adds key value pair "two" -> 2
let hasOne = dict.contains("one") //will return true since key "one" exists
let hasThree = dict.contains("three") //will return false since key "theee" doesn't exists

//tries to get the value from the dictionary with key "one"
select one_as_int from dict.try_get_value("one")
  print(one_as_int) //will print "1" as value to key "one" is 1
