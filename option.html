Option

Option is libary used to reduce null checking.

Constructors

Option<T> T value => Option
Some T value => Option
None<T> => Option

Functions

is_none => bool
is_some => bool
value_or T other_value => T
to_string => string

Examples

fn get_optional_value int num => Option
  return num % 2 == 0 ? Some(num) : None()
;

let a_opt = get_optional_value(1) //return None
let b_opt = get_optional_value(2) //return Some(2)

select a from a_opt do print(a) //will not print cause it's None
select b from b_opt do print(b) //prints "2"

if a_opt.is_none() do print("has nothing") //prints "has nothing"
if b_opt.is_some() do print("has something") //prints "has something"

print(a_opt.value_or(0)) //prints "0"
print(b_opt.value_or(0)) //prints "2"

print(a_opt.to_string()) //prints "None"
print(b_opt.to_string()) //prints "Some(2)"
