//adv Operators are =&= , =|= , =!= , =>= , =<= , <<< , >>>
let a = 1

//ALL EQUALS
if a =&= 1 : 1 : 1 do return

//ANY EQUALS
if a =|= 3 : 4 : 1 do return

//ALL NOT EQUALS
if a =!= 2 : 3 : 4 do return

//ALL EQUALS OR MORE
if a =>= 1 : 3 : 4 do return

//ALL EQUALS OR LESS
if a =<= 1 : 0 : -1 do return

//ALL LESS
if a <<< 0 : -1 : -34 do return

//ALL MORE
if a >>> 3 : 34 : 4 do return
