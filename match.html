Match

Match is similar to switch but can be used for other values than constants.

let a = "ZERO"
match a
  "ZERO" => do println(0)
  "ONE" => do println(1)
  _ => do println("Unkown")
;

let b = match a
  "ZERO" => do 0 //if "ZERO" b = 0
  "ONE" => do 1 //elif "ONE" b = 1
  _ => // else b = -1
    println("Unknown")
    -1
  ;
;

a = match b
  0 => do "ZERO" //if 0 a = "ZERO"
  1 => do "ONE" //elif 1 a = "ONE"
  _ => // else a = "MINUS ONE"
    println("Unknown")
    "MINUS ONE"
  ;
;

//you can also use operators to override the match operator
match a
   > 0 => do println("more than zero")
   < 0 => do println("more than zero")
   != 0 => do println("not zero")
   ....
   0 => do println("zero")
;

//you can even use member expressions to override the operator
let a_opt = Some(a)
match a_opt
   .is_some() => do println("some")
   _ => do println("none")
;

match &a_opt
   ->is_some() => do println("some")
   _ => do println("none")
;

Example

Using Match:

let b = match a
  "a" => do 1
  "b" => do 2
  "c" => do 3
  "d" => do 4
  _ => do -1
;

Using basic syntax:

int b = 0
if a == "a" do b = 1
elif a == "b" do b = 2
elif a == "c" do b = 3
elif a == "d" do b = 4
else do b = -1
