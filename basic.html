Using Options

get => print, option

fn main =>
  let opt = Some(0)
  select value from opt do print(value) //prints zero
  let opt = None()
  select value from opt do print(value) //doesn't print because it's none
  print(opt.value_or(10)) //prints 10
  if opt.is_none()) do print("none") //will print none
;
