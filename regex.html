re string line, string reg, int capture => vector<RegexMatch>

Examples

let matches = re("Hello text world", "text", 0)
foreach m in matches do print($"Pos: {m.position} Text: {m.match}")
//will print "Pos: 6 Text: text"

//or with precompiled regex
regex reg("text")
let matches = re("Hello text world", reg, 0)
foreach m in matches do print($"Pos: {m.position} Text: {m.match}")
//will print "Pos: 6 Text: text"
