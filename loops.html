Loops

for/foreach/while/loop

Examples

// prints 0-9
for i until 10
  println(i)
;
// or
for i until 10 do println(i)

// prints "hello world!"
vector<string> lines = {"hello ", "world", "!"}
foreach i, line in lines
  print(lines)
;
// prints "0-2"
foreach i, line in lines do println(i)

// prints 0-9
let i = 0
while i < 10
  println(i++)
;
// prints 0-9

let i = 0
loop
  println(i++)
  if i < 10 do break
;
