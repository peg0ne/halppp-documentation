Map

functional style map

Examples

struct Int
  int num
  string num_name
  Int int in_num =>
    num = in_num
    num_name = $"{in_num}"
  ;
;

// collects a vector of numbers up to 100000
let a = linq::range(100000).as_vec()

let b = map<string>(a)
  .where(x => x > 10)
  .where(x => x < 1000)
  .where(x => x % 2 == 0)
  .select(x => Int(x))
  .where(x => x.num_name[0] == '2')
  .where(x => x.num % 3 == 0)
  .select(x => x.num_name)

//b now contains these numbers as strings
//["24", "204", "210", "216", "222",
// "228", "234", "240", "246",
// "252", "258", "264", "270",
// "276", "282", "288", "294"]
